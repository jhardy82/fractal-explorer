## T12: Crosshair Coordinate Display

**Commit message:** `feat(ui): crosshair coordinate display — complex coords under mouse cursor`  
**Test file:** `tests/test_crosshair.py`  
**Model:** Haiku implementer

### Design

When hovering over an `EscapeTimeFractal` page, show the complex-plane coordinate of the
mouse position in the lower-left of the body area (just above the nav bar). Passive display
— no toggle needed, always on for escape-time pages. Implemented as a module-level helper
`_screen_to_complex()` plus a method `_format_coord()` on `FractalExplorer`.

### Changes to `src/fractal_explorer_v2.py`

**1. Module-level helper — add after `_gradient_palette` (near line 124):**

```python
def _screen_to_complex(
    mx: int, my: int,
    x_range: tuple[float, float],
    y_range: tuple[float, float],
    w: int,
    body_h: int,
    body_y_offset: int,
) -> complex:
    """Map screen pixel (mx, my) to the complex number at that position."""
    body_y = max(0, min(my - body_y_offset, body_h - 1))
    re = x_range[0] + (mx / w) * (x_range[1] - x_range[0])
    im = y_range[0] + (body_y / body_h) * (y_range[1] - y_range[0])
    return complex(re, im)
```

**2. `FractalExplorer` method — add after `reset_current()` (near line 1844):**

```python
def _format_coord(self, page: EscapeTimeFractal) -> str:
    mx, my = pygame.mouse.get_pos()
    z = _screen_to_complex(mx, my, page.x_range, page.y_range, self.w, self.body_h, TITLE_H)
    sign = "+" if z.imag >= 0 else "-"
    return f"{z.real:+.6f} {sign} {abs(z.imag):.6f}i"
```

**3. `draw_chrome()` — add at the end (before the closing `# nav bar` comment or after last blit):**

```python

if isinstance(self.current, EscapeTimeFractal):
    coord_surf = self.font_xs.render(self._format_coord(self.current), True, DIM)
    self.screen.blit(coord_surf, (8, self.h - NAV_H - 16))
```

```python
"""test_crosshair.py — T12 crosshair coordinate display tests."""
from fractal_explorer_v2 import _screen_to_complex

class TestScreenToComplex:
    def test_left_top_maps_to_x_range_start(self):
        z = _screen_to_complex(0, 30, (-2.5, 1.0), (-1.25, 1.25), 1600, 814, 30)
        assert abs(z.real - (-2.5)) < 1e-9
        assert abs(z.imag - (-1.25)) < 1e-9

    def test_right_edge_maps_to_x_range_end(self):
        z = _screen_to_complex(1600, 30, (-2.5, 1.0), (-1.25, 1.25), 1600, 814, 30)
        assert abs(z.real - 1.0) < 0.01

    def test_center_maps_to_range_midpoint(self):
        z = _screen_to_complex(800, 30 + 407, (-2.5, 1.0), (-1.25, 1.25), 1600, 814, 30)
        assert abs(z.real - (-0.75)) < 0.01   # (-2.5 + 1.0) / 2
        assert abs(z.imag - 0.0) < 0.01        # (-1.25 + 1.25) / 2

    def test_clamps_mouse_above_body(self):
        # my < body_y_offset → clamped to body_y=0 → imag = y_range[0]
        z = _screen_to_complex(0, 0, (-2.5, 1.0), (-1.25, 1.25), 1600, 814, 30)
        assert z.imag == -1.25

    def test_known_point(self):
        # mx=400 (25% of 1600) → re = -2.5 + 0.25 * 3.5 = -1.625
        z = _screen_to_complex(400, 30, (-2.5, 1.0), (-1.25, 1.25), 1600, 814, 30)
        assert abs(z.real - (-1.625)) < 1e-9
```

## T13: Zoom Stack / Location Bookmarks

**Commit message:** `feat(interaction): zoom stack — B saves viewport, N cycles through bookmarks`  
**Test file:** `tests/test_zoom_stack.py`  
**Model:** Sonnet implementer (stateful, multiple transitions)

### Design

B key saves the current escape-time viewport (form identity + x/y ranges). N key cycles
through saved bookmarks in order, restoring each viewport. Ring buffer capped at 10. Chrome
shows bookmark count in the title bar centre when non-empty.

### Changes to `src/fractal_explorer_v2.py`

**1. `FractalExplorer.__init__` — add after `self._cinematic = False` (line ~1804):**


**2. `handle_event()` KEYDOWN block — add after `elif k == pygame.K_s:` block:**

```python
elif k == pygame.K_b:
    page = self.current
    if isinstance(page, EscapeTimeFractal):
        entry = (self.cat_idx, self.page_idx, page.x_range, page.y_range)
        self._bookmarks.append(entry)
        if len(self._bookmarks) > 10:
            self._bookmarks.pop(0)
        self._bookmark_idx = len(self._bookmarks) - 1
elif k == pygame.K_n:
    if self._bookmarks:
        self._bookmark_idx = (self._bookmark_idx + 1) % len(self._bookmarks)
        ci, pi, xr, yr = self._bookmarks[self._bookmark_idx]
        self.cat_idx = ci
        self.page_idx = pi
        page = self.current
        if isinstance(page, EscapeTimeFractal):
            page.x_range = xr
            page.y_range = yr
            page.row = 0
```

```python
"""test_zoom_stack.py — T13 zoom stack / bookmark tests."""
import pygame
import pytest

def _make_explorer(engine):
    if not pygame.get_init():
        pygame.init()
    if pygame.display.get_surface() is None:
        pygame.display.set_mode((10, 10))
    return engine.FractalExplorer()

class TestZoomStack:
    def test_b_key_saves_bookmark(self, engine):
        ex = _make_explorer(engine)
        ex.cat_idx = 0; ex.page_idx = 0
        assert isinstance(ex.current, engine.Mandelbrot)
        ev = pygame.event.Event(pygame.KEYDOWN, key=pygame.K_b, mod=0, unicode='b')
        ex.handle_event(ev)
        assert len(ex._bookmarks) == 1

    def test_bookmark_captures_viewport(self, engine):
        ex = _make_explorer(engine)
        ex.cat_idx = 0; ex.page_idx = 0
        page = ex.current
        page.x_range = (-1.0, 0.5)
        page.y_range = (-0.5, 0.5)
        ev = pygame.event.Event(pygame.KEYDOWN, key=pygame.K_b, mod=0, unicode='b')
        ex.handle_event(ev)
        ci, pi, xr, yr = ex._bookmarks[0]
        assert xr == (-1.0, 0.5)
        assert yr == (-0.5, 0.5)

    def test_n_key_restores_bookmark(self, engine):
        ex = _make_explorer(engine)
        ex.cat_idx = 0; ex.page_idx = 0
        page = ex.current
        page.x_range = (-1.0, 0.5)
        page.y_range = (-0.5, 0.5)
        b_ev = pygame.event.Event(pygame.KEYDOWN, key=pygame.K_b, mod=0, unicode='b')
        ex.handle_event(b_ev)
        page.x_range = (-2.5, 1.0)
        n_ev = pygame.event.Event(pygame.KEYDOWN, key=pygame.K_n, mod=0, unicode='n')
        ex.handle_event(n_ev)
        assert ex.current.x_range == (-1.0, 0.5)

    def test_ring_buffer_caps_at_10(self, engine):
        ex = _make_explorer(engine)
        ex.cat_idx = 0; ex.page_idx = 0
        ev = pygame.event.Event(pygame.KEYDOWN, key=pygame.K_b, mod=0, unicode='b')
        for _ in range(12):
            ex.handle_event(ev)
        assert len(ex._bookmarks) == 10

    def test_n_on_empty_is_noop(self, engine):
        ex = _make_explorer(engine)
        prev_cat = ex.cat_idx
        ev = pygame.event.Event(pygame.KEYDOWN, key=pygame.K_n, mod=0, unicode='n')
        ex.handle_event(ev)
        assert ex.cat_idx == prev_cat

    def test_b_noop_on_non_escape_time(self, engine):
        ex = _make_explorer(engine)
        ex.cat_idx = CAT_KEYS.index('B') if hasattr(engine, 'CAT_KEYS') else 1
        ev = pygame.event.Event(pygame.KEYDOWN, key=pygame.K_b, mod=0, unicode='b')
        ex.handle_event(ev)
        # IFS page → bookmark list stays empty
        assert len(ex._bookmarks) == 0 or all(
            not isinstance(ex.pages.get(engine.CAT_KEYS[ci], [])[pi] if hasattr(engine, 'CAT_KEYS') else [], engine.EscapeTimeFractal)
            for ci, pi, *_ in ex._bookmarks
        )
```

## T14: Math Info Panel

**Commit message:** `feat(ui): math info panel — M key shows formula overlay for current fractal`  
**Test file:** `tests/test_info_panel.py`  
**Model:** Haiku implementer

### Design

Each `FractalPage` gains a `formula: str = ""` class attribute. For key escape-time forms,
fill it with a human-readable mathematical formula. M key toggles `_show_info`. When on,
`draw_chrome()` renders a semi-transparent SRCALPHA overlay in the upper-left of the body
area showing the formula and the existing `page.info` string.

```python
if self._show_info:
    page = self.current
    lines = []
    if page.formula:
        lines.append(page.formula)
    if page.info:
        lines.append(page.info)
    if lines:
        pad = 8
        line_h = 14
        overlay_w = max(self.font_xs.size(ln)[0] for ln in lines) + pad * 2
        overlay_h = len(lines) * line_h + pad * 2
        overlay = pygame.Surface((overlay_w, overlay_h), pygame.SRCALPHA)
        overlay.fill((8, 11, 20, 210))
        for i, ln in enumerate(lines):
            txt = self.font_xs.render(ln, True, FG)
            overlay.blit(txt, (pad, pad + i * line_h))
        self.screen.blit(overlay, (8, TITLE_H + 8))
```
